void put(int *scale1, int *scale2, int width, int height, t_data *game)
{
 void *img;
 int *addr;
 int i;
 int j;
 int k;
  img = mlx_new_image(game->mlx, width, height);
  addr = (int *)mlx_get_data_addr(img, &game->textures->bpp, &game->textures->size_line, &game->textures->endian);
  i = 0;
 while(i < height / 2)
 {
	j = 0;
   while(j < width)
   {
	 k = width * i + j;
	 addr[width * i + j] = scale1[k];
	 j++;
   }
   i++;
 }
 i = height / 2;
 while(i < height)
 {
	j = 0;
   while(j < width)
   {
	 k = width * i + j;
	 addr[width * i + j] = scale2[k];
	 j++;
   }
   i++;
 }
  mlx_put_image_to_window(game->mlx, game->win, img, 0, 0);
  mlx_destroy_image(game->mlx, img);
  mlx_loop(game->mlx);
}
